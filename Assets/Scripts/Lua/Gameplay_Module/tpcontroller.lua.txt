local UnityEngine = CS.UnityEngine
local Debug = CS.UnityEngine.Debug
local Log = Debug.Log
local LogError = Debug.LogError
local Vector3 = UnityEngine.Mathf.Vector3

TPChController = {
    tpCharactor = nil,
    tpCameraPivot = nil,
    tpCamera = nil
}
TPChController.__index = TPChController

function TPChController:new(tpCharactor, tpCameraPivot, tpCamera)
    local result = {}
    result.tpCharactor = tpCharactor
    result.tpCameraPivot = tpCameraPivot
    result.tpCamera = tpCamera
    setmetatable(result, self)
    return result
end

function TPChController:start()
    Debug.Log("tp controller start")
end

function TPChController:update()
    Debug.Log("tp controller update")
end

function TPChController:destroy()
    Debug.Log("tp controller destroy")
end
